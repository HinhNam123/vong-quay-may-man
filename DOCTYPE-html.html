/**
* Initializes the prize inventory in Firestore if it doesn't exist.
* Sets up a real-time listener (onSnapshot) for prize updates.
*/
async function initializePrizes() {
const prizesRef = collection(db,
`artifacts/${appId}/public/data/prizes`);
const prizeDocRef = doc(prizesRef, 'inventory');
const prizeDoc = await getDoc(prizeDocRef);
// If prize inventory doesn't exist, create it with initial data
if (!prizeDoc.exists()) {
const prizeData = {};
initialPrizes.forEach(p => { prizeData[p.id] = p; });
await setDoc(prizeDocRef, { all: prizeData });
console.log("Initial prizes set in Firestore.");
}
// Listen for real-time updates to the prize inventory
onSnapshot(prizeDocRef, (doc) => {
const prizeData = doc.data().all;
// Sort prizes by their defined order
prizes = Object.values(prizeData).sort((a, b) => a.order -
b.order);
updatePrizeList(); // Update the displayed prize list
createWheel(); // Re-create the wheel with updated prizes
});
}

/**
* Handles the share action. Grants an additional turn if the user
hasn't shared before.
* Updates the user's document in Firestore.
*/
async function handleShare() {
if (userState.hasShared || !userState.phone) {
showSimpleModal('Thng bo', 'Bn nhn thng chia s hoc
cha ng k.');
return;
}
shareBtn.disabled = true; // Disable button immediately
shareBtn.textContent = 'ang x l...';
const userDocRef = doc(db, `artifacts/${appId}/public/data/users`,
userState.phone);
try {
await runTransaction(db, async (transaction) => {
const userDoc = await transaction.get(userDocRef);
if (!userDoc.exists()) {
throw "User document not found.";
}
const userData = userDoc.data();
if (userData.hasShared) {
throw "User has already received share bonus.";
}
// Update user's turns and set hasShared to true
transaction.update(userDocRef, {
turns: (userData.turns || 0) + 1,
hasShared: true
});
});
userState.hasShared = true;
userState.turns++;
updateUI();
showSimpleModal('Thnh Cng!', 'Cm n bn chia s. Bn
c tng thm 1 lt quay!');
} catch (error) {
console.error("Error handling share:", error);
showSimpleModal("Li", "C li xy ra khi chia s. Vui lng
th li.");
userState.hasShared = false; // Revert state if transaction
fails
} finally {
updateUI(); // Ensure UI is updated after attempt
}
}
/**
* Handles the spinning of the wheel.
* Uses a Firestore transaction to decrement prize quantity and user's
turns.
* Animates the wheel and displays the winning prize.
*/
async function spinWheel() {
if (userState.turns <= 0 || spinBtn.dataset.spinning === 'true') {
showSimpleModal('Thng bo', 'Bn khng cn lt quay hoc
vng quay ang hot ng.');
return;
}
spinBtn.dataset.spinning = 'true'; // Set spinning flag
updateUI(); // Disable spin button
try {
const winningPrize = await runTransaction(db, async
(transaction) => {
// Read all necessary documents first
const prizesRef = doc(db,
`artifacts/${appId}/public/data/prizes`, 'inventory');
const prizesDoc = await transaction.get(prizesRef);
if (!prizesDoc.exists()) {
throw "Error: Prize inventory not found.";
}
const userDocRef = doc(db,
`artifacts/${appId}/public/data/users`, userState.phone);
const userDoc = await transaction.get(userDocRef);
if (!userDoc.exists()) {
throw "User document not found during spin.";
}
let currentPrizesData = prizesDoc.data().all;
// Filter for prizes that still have quantity > 0
let availablePrizes =
Object.values(currentPrizesData).filter(p => p.quantity > 0);
if (availablePrizes.length === 0) {
showSimpleModal('Rt Tic!', 'Tt c phn qu c
pht ht.');
return null; // No available prizes
}
// Select a random prize from available ones
const randomPrize =
availablePrizes[Math.floor(Math.random() * availablePrizes.length)];
currentPrizesData[randomPrize.id].quantity--; // Decrement
quantity of the chosen prize
// Now perform all writes
transaction.update(prizesRef, { all: currentPrizesData });
// Update prize inventory in Firestore
const userData = userDoc.data();
const newTurns = (userData.turns || 0) - 1;
const newWins = [...(userData.wins || []), {
prizeId: randomPrize.id,
prizeName: randomPrize.name,
timestamp: new Date()
}];
transaction.update(userDocRef, { turns: newTurns, wins:
newWins });
return randomPrize; // Return the winning prize
});
if (winningPrize) {
// Calculate rotation for the wheel animation
const winningSegmentIndex = prizes.findIndex(p => p.id ===
winningPrize.id);
const segmentAngle = 360 / prizes.length;
// Add a small random offset for more natural spin
const randomOffset = (Math.random() - 0.5) * segmentAngle
* 0.8;
// Calculate target angle to stop at the winning segment
const targetAngle = (360 - (winningSegmentIndex *
segmentAngle)) - (segmentAngle / 2) + randomOffset;
const totalRotation = 360 * 5 + targetAngle; // Spin 5
full rotations plus target angle
wheel.style.transform = `rotate(${totalRotation}deg)`;
// After the spin animation completes
setTimeout(() => {
userState.turns--; // Decrement local user turns
showResultModal(winningPrize.name); // Show result
modal
spinBtn.dataset.spinning = 'false'; // Reset spinning
flag
updateUI(); // Update UI
// Reset wheel rotation for the next spin without
animation
const currentRotation = totalRotation % 360;
wheel.style.transition = 'none'; // Temporarily
disable transition
wheel.style.transform =
`rotate(${currentRotation}deg)`;
wheel.offsetHeight; // Trigger reflow to apply instant
change
wheel.style.transition = 'transform 5s ease-out'; //
Re-enable transition
}, 5500); // Duration matches CSS transition
} else {
spinBtn.dataset.spinning = 'false';
updateUI();
}
} catch (error) {
console.error("Transaction failed: ", error);
showSimpleModal("Li", "C li xy ra khi quay. Vui lng th
li.");
spinBtn.dataset.spinning = 'false';
updateUI();
}
}
